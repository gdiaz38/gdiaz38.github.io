import pandas as pd import numpy as np from datetime import datetime, timedelta # Load and process sales data def analyze_sales_performance(df): # Calculate key metrics total_sales = df['sales_amount'].sum() avg_deal_size = df['sales_amount'].mean() conversion_rate = len(df[df['status'] == 'won']) / len(df) # Regional performance analysis regional_performance = df.groupby('region').agg({ 'sales_amount': ['sum', 'mean', 'count'], 'salesperson': 'nunique' }).round(2) # Time series analysis df['month'] = pd.to_datetime(df['date']).dt.to_period('M') monthly_trends = df.groupby('month')['sales_amount'].sum() growth_rate = monthly_trends.pct_change().mean() * 100 # Top performers identification top_performers = df.groupby('salesperson')['sales_amount'].sum() top_performers = top_performers.sort_values(ascending=False).head(5) return { 'total_sales': total_sales, 'avg_deal_size': avg_deal_size, 'conversion_rate': conversion_rate, 'growth_rate': growth_rate, 'top_performers': top_performers } # Predictive analytics def forecast_sales(historical_data, periods=3): from sklearn.linear_model import LinearRegression X = np.array(range(len(historical_data))).reshape(-1, 1) y = historical_data.values model = LinearRegression() model.fit(X, y) future_X = np.array(range(len(X), len(X) + periods)).reshape(-1, 1) predictions = model.predict(future_X) return predictions
